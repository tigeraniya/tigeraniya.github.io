@receiver(post_save, sender=settings.AUTH_USER_MODEL)
def create_auth_token(sender, instance=None, created=False, **kwargs):
    if created:
        Token.objects.create(user=instance)

    [docs]class Job(models.Model):
    name = models.CharField(max_length=50)
    desc = models.CharField(max_length=100)
    created = models.DateTimeField(auto_now_add=True)
    responses = models.IntegerField(default=0)
    user = models.ForeignKey(User, related_name='job')
    completed = models.BinaryField(default=False)
    lat = models.CharField(default='')
    lon = models.CharField(default='')
    cat = models.CharField(default='')
    price_range = models.ForeignKey(PriceRange)

    def __unicode__(self):
        return "<Job :%s>" % self.name
